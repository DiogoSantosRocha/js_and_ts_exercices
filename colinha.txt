ls > liste os arquivos da pasta
mkdir nome > crie a pasta nome
rm -r nome > apaga o arquivo nome

git commit -m "mensagem" > finaliza um sessão de trabalho 
git push > manda alterações para o github
git pull > puxa alaterações do github
